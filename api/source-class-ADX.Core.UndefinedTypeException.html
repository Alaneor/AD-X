301 /**
302  * This exception ( or its more specific subclass ) is thrown when any of php's ldap functions triggers errors
303  */
304 class LdapNativeException extends Exception
305 {
306     public function __construct( $link_id )
307     {
308         parent::__construct();
309 
310         ldap_get_option( $link_id, LDAP_OPT_HOST_NAME, $domain );
311 
312         $domain         = explode( ':', $domain )[0]; // Get the server name and discard the port
313         $this->code     = ldap_errno( $link_id );
314         $this->message  = "$domain: ".ldap_error( $link_id );
315     }
316 }
