 26 /**
 27  * Provides conversion functionality between ldap and php datatypes
 28  *
 29  * All the conversion functionality is defined here. This class is used
 30  * internally to convert data automatically to php as it is received from
 31  * ldap server and also to convert data to ldap when data is being written

 51     protected static function _convert( $direction, $attribute, $values )
 52     {
 53         // Load the schema information for the current attribute
 54         $schema     = Schema::get( "$attribute" );
 55         $ats        = $schema['attributesyntax'][0];
 56         $oms        = $schema['omsyntax'][0];
 57 
 58         $values     = (array)$values;
 59         $converted  = array();
 60         $method     = null;
 61         $class      = get_called_class();
 62 
 63         // Based on syntax, choose the corresponding syntax conversion method
 64         switch ( $ats )
 65         {
 66             case Syntax::Binary:
 67                 $method = 'binary';
 68                 break;
 69 
 70             case Syntax::Boolean:
 71                 $method = 'bool';
 72                 break;
 73 
 74             case Syntax::DnString:
 75                 $method = 'object';
 76                 break;
 77 
 78             case Syntax::Integer:
 79             case Syntax::LargeInt:
 80                 // Conversion not needed
 81                 break;
 82 
 83             case Syntax::UnicodeString:
 84             case Syntax::TeletexString:
 85             case Syntax::PrintableString:
 86             case Syntax::NumericString:
 87                 // Conversion not needed
 88                 break;
 89 
 90             case Syntax::Time:
 91                 switch ( $oms )
 92                 {
 93                     case '23':  // UTC Time format
 94                         $method = 'utctime';
 95                         break;
 96 
 97                     case '24':  // Generalised Time format
 98                         $method = 'generalisedtime';
 99                         break;
100                 }
101                 break;
102         }
103 
104         // Define some syntax overrides based on attribute names ( a hit here will replace
105         // the result from the previous switch )
106         switch ( "$attribute" )
107         {
108             case 'currenttime':                 // Not in schema, present on rootDSE
109                 $method = 'generalisedtime';
110                 break;
111 
112             case 'issynchronized':              // Not in schema, present on rootDSE
113             case 'isglobalcatalogready':        // Not in schema, present on rootDSE
114                 $method = 'bool';
115                 break;
116 
117             case 'unicodepwd':                  // A password always needs special treatment
118             case 'objectguid':                  // I want objectguid shown as AD tools show it ( like Powershell )
119                 $method = $attribute;
120                 break;
121 
122             // These attributes have LargeInt as syntax, but their meaning is different ( they represent a time )
123             case 'pwdlastset':
124             case 'accountexpires':
125             case 'lastlogon':
126             case 'lockouttime':
127                 $method = 'timestamp';
128                 break;
129         }
130 
131         // If a conversion method has been found, call it for all values in $value
132         $method = '_'.$direction.'_'.$method;
133 
134         if ( is_callable( [$class, $method]) )
135         {
136             foreach ( $values as $value ) $converted[] = call_user_func( [ $class, $method], $value );
137         }
138         else $converted = $values;  // No conversion has been found - use the raw data
139 
140         // Aaand we are done!
141         return $converted;
142     }

146     protected static function _to_l_timestamp( $timestamp )
147     {
148         if ( $timestamp === 0 ) return 0;
149 
150         return ( ( $timestamp * 10000000 ) - 11644473600 );
151     }

153     protected static function _to_p_timestamp( $timestamp )
154     {
155         // The second part of the conditiona takes care of some crazy behaviour
156         // ( documented, though ) of accountExpires, which can be zero
157         // or 0x7FFFFFFFFFFFFFFF when the account is set to never expire.
158         // Hopefully, this will not have any side effects...
159         if ( $timestamp === 0 || $timestamp === hexdec('0x7FFFFFFFFFFFFFFF') ) return 0;
160 
161         return ( floor( $timestamp / 10000000 ) + 11644473600 );
162     }

223     protected static function _to_l_unicodepwd( $password )
224     {
225         $password   = "\"$password\"";  // Enclose the password in double quotes
226 
227         $length     = strlen( $password );
228         $pwd        = '';
229 
230         for ( $i = 0; $i < $length; $i++ )
231         {
232             $pwd .= "{$password{$i}}\000";  // Pad every character with a NULL value ( null-padded string )
233         }
234 
235         return $pwd;
236     }

238     protected static function _to_p_objectguid( $guid )
239     {
240         // An interesting piece of code I have found that converts
241         // the objectguid exactly to what standard Powershell or other MS-based
242         // tools show.
243         $hex_guid = unpack( "H*hex", $guid );
244         $hex    = $hex_guid["hex"];
245 
246         $hex1   = substr( $hex, -26, 2 ) . substr( $hex, -28, 2 ) . substr( $hex, -30, 2 ) . substr( $hex, -32, 2 );
247         $hex2   = substr( $hex, -22, 2 ) . substr( $hex, -24, 2 );
248         $hex3   = substr( $hex, -18, 2 ) . substr( $hex, -20, 2 );
249         $hex4   = substr( $hex, -16, 4 );
250         $hex5   = substr( $hex, -12, 12 );
251 
252         $guid = $hex1 . "-" . $hex2 . "-" . $hex3 . "-" . $hex4 . "-" . $hex5;
253 
254         return $guid;
255     }

257     protected static function _to_l_objectguid( $guid )
258     {
259         $guid = str_replace( '-', '', $guid );
260 
261         $octet_str  = substr( $guid, 6,     2 );
262         $octet_str .= substr( $guid, 4,     2 );
263         $octet_str .= substr( $guid, 2,     2 );
264         $octet_str .= substr( $guid, 0,     2 );
265         $octet_str .= substr( $guid, 10,    2 );
266         $octet_str .= substr( $guid, 8,     2 );
267         $octet_str .= substr( $guid, 14,    2 );
268         $octet_str .= substr( $guid, 12,    2 );
269         $octet_str .= substr( $guid, 16,    strlen( $guid ) );
270 
271         $hex_guid = '';
272 
273         for ( $i = 0; $i <= strlen( $octet_str ) - 2; $i = $i + 2 )
274         {
275             $hex_guid .=  "\\" . substr( $octet_str, $i, 2 );
276         }
277 
278         return $hex_guid;
279     }
