 26 /**
 27  * Performs lookup operations on ldap server
 28  *
 29  * You can use Task to perform searches on the ldap server according to options you
 30  * specify.<br>
 31  * This class serves as a wrapper for php functions that retrieve and process
 32  * data from directory servers.
 33  *
 34  * <h2>Features:</h2>
 35  * <h3>Method Chaining</h3>
 36  * Methods may be chained together to simplify configuration. Ensure that the methods you wish to chain
 37  * together return the {@link self} object to avoid issues. See the example code below.<br>
 38  * <br>
 39  * <p class="alert">Make sure you import the class into current namespace before trying to use it:<br>
 40  * <code>use ADX\Core\Task;</code><br>
 41  * To simplify the examples, the above line is not present in the examples but it is assumed you have
 42  * it in your implementation.</p>
 43  *
 44  * <h2>Example:</h2>
 45  * <code>
 46  * use ADX\Enums;                           // Import the Enums namespace
 47  *
 48  * $link = new Link( 'example.com' );               // Connect to server
 49  * $link->bind( 'user@example.com', 'MySecretPwd' );        // Authenticate
 50  *
 51  * $task = new Task( Enums\Operation::OpSearch, $link );        // Create a search operation
 52  * $task    ->get_attributes( ['cn', 'givenname', 'sn', 'mail'] )   // Load these attributes for the objects
 53  *  ->set_filter( '(&(objectclass=user)(mail=*))' )     // Get all users that have a mail attribute
 54  *  ->set_base( 'OU=Users,DC=example,DC=com' );     // Limit the search only to the Users OU in the domain
 55  *
 56  * // Get the results!
 57  * $result = $task->run();
 58  * </code>
 59  *
 60  * @see         {@link Link} - Use the Link object to establish and configure connection to directory servers
 61  * @see         {@link Enums\Operation} - Specify the scope of the lookup operation
 62  * @see         <a href="http://msdn.microsoft.com/en-us/library/windows/desktop/ms678001%28v=vs.85%29.aspx">MSDN - Deciding Where To Search</a>
 63  *
 64  * @todo        Move all lookup error checking from {@link Object} to {@link self}
 65  * @todo        The connection pooling feature might need a rewrite - if I start a lookup operation with an account that has lower access rights than the one I used previously, AD-X might fail to perform the requested operation.
 66  */

144      *
145      * This allows you to retrieve complete search results for queries that would otherwise

228     /**
229      * Perform the lookup operation on server and return the resultset
230      *
231      * This method sends the lookup request to the server with the
232      * configuration you provided. If the {@link Link} has pagination enabled,
233      * it will return a single page. To get the subsequent pages, call this method
234      * again on the object, without modifying the lookup criteria.
235      *
236      * @uses        Link::$rootDSE      to get the domain base dn ( from <i>defaultnamingcontext</i> ) if no override has been specified
237      * @return      Result|false        The Result object with returned Objects or FALSE if maximum number of referrals was chased
238      */
239     public function run()
240     {
241         // Prepare the information needed for the operation
242         $link_id    = $this->adxLink->get_link();
243         $baseDN     = ( $this->dn || $this->dn === '' ) ? $this->dn : $this->adxLink->rootDSE->defaultnamingcontext->value( 0 );
244         if ( $this->page_size && $this->complete ) $this->cookie = '';  // Reset the cookie if the operation is started again
245         $this->complete = false;                                        // Reset the completion status
246 
247         // Send the pagination control cookie if present
248         // I must check explicitly for NULL bcause '' also evaluates to FALSE
249         if ( $this->cookie !== null ) ldap_control_paged_result( $link_id, $this->page_size, true, $this->cookie );
250 
251         // Perform the lookup operation
252         // All exceptions thrown here will bubble up!
253         $result_id = $this->_directory_lookup( $this->operationType, $link_id, $baseDN, $this->filter, $this->attributes );
254 
255         // Get the new cookie from the response if pagination is enabled
256         // I must check explicitly for NULL bcause '' also evaluates to FALSE
257         if ( $this->cookie !== null ) ldap_control_paged_result_response( $link_id, $result_id, $this->cookie );
258 
259         // Lookup operation successful - continue processing...
260         $referral_link = $this->_parse_referrals( $result_id );
261 
262         // Check for referrals
263         if ( $referral_link )                                       // We have been referred to a new Link
264         {
265             if ( $this->referral_counter <= $this->max_referrals )  // Are we still allowed to chase referrals?
266             {
267                 // Do we have this connection in the pool already?
268                 if ( isset( static::$connection_pool[$referral_link] ) )
269                 {
270                     $link = static::$connection_pool[$referral_link];   // Use the existing link to perform the lookup
271                 }
272                 else
273                 {
274                     $link = $this->adxLink->_redirect( $referral_link );    // Reconnect to the new server
275                     static::$connection_pool[$referral_link] = $link;       // Save the connection in the link pool
276                 }
277 
278                 $this->referral_counter++;
279                 $this->adxLink = $link;                                     // Replace the current link with the new link
280 
281                 return $this->run();    // Only those calls that actually succeed in getting data should continue with code below
282             } else return false;        // Reached maximum number of referrals -> do not continue
283         }
284 
285         // Successfully retrieved data from server
286         $this->referral_counter = 0;
287 
288         $result = ldap_get_entries( $this->adxLink->get_link(), $result_id );
289 
290         unset( $result['count'] );  // Clean up some unneeded mess ( more cleanup happens at object instantiation )
291 
292         $data = array();
293 
294         // Create new objects from result, converting the values to php-compatible data formats on the way
295         foreach ( $result as $objectData ) $data[] = new Object( $this->adxLink, $objectData, true );
296 
297         $this->complete = ! $this->cookie;  // As long as cookie is present, the result cannot be complete
298 
299         return new Result( $data );
300     }

302     /**
303      * Return all pages at once when doing paged search operations
304      *
305      * Use this method to get a complete resultset with all pages at once.
306      * Note that pagination must be enabled first in order for this
307      * method to do something more than {@link self::run()}. You can
308      * enable it by calling {@link self:use_pages()} or by passing a
309      * number of objects per page into this method.
310      *

312      * @return      Result      A Result object containing the objects on the server
313      *
314      * @see         self::use_pages()
315      */
316     public function run_paged( $page_size = null )
317     {
318         if ( isset( $page_size ) ) $this->use_pages( $page_size );
319 
320         $resultset = array();
321 
322         do
323         {
324             $data = $this->run();
325 
326             if ( $data instanceof Result )
327             {
328                 $resultset = array_merge( $resultset, $data->to_array() );
329             }
330             else return false;
331         }
332         while ( ! $this->complete );
333 
334         return new Result( $resultset );
335     }

338     protected function _parse_referrals( $result_id )
339     {
340         if ( ! ldap_parse_result( $this->adxLink->get_link(), $result_id, $code, $matchedDN, $errMsg, $referrals ) ) return;
341 
342         if ( count( $referrals ) == 0 ) return;
343 
344         // Referral looks like this:
345         // ldap://example.com/DC=example,DC=com
346         //
347         // Extract the protocol and url from the referral string
348         if ( preg_match( "/^ldaps?:\/\/[a-zA-Z0-9_\-.]*/", $referrals[0], $matches ) !== 1 ) throw new Exception( "Unable to parse referral - {$referrals[0]}" );
349 
350         // Match found? Return it!
351         if ( isset( $matches[0] ) ) return $matches[0];
352     }

355     protected function _directory_lookup( $operation, $link_id, $dn, $filter, $attributes )
356     {
357         // Ensure that attributes required by the library to work correctly are always present in the resultset
358         $mandatory = [
359             'objectclass',
360         ];
361 
362         $attributes = array_merge( $attributes, $mandatory );
363 
364         // php's internal mechanisms should take care of that... Keeping this here only for
365         // a possible future reference.
366         // $attributes = array_map( 'strtolower', $attributes );
367         // $attributes = array_unique( $attributes );
368 
369         // Perform the operation
370         $result_id = $operation( $link_id, $dn, $filter, $attributes );
371 
372         if ( $result_id === false ) // Operation was unsuccessful
373         {
374             // Check for any possible errors and handle them ( probably a TODO )
375             $code = ldap_errno( $link_id );
376 
377             switch( $code )
378             {
379                 case Enums\ServerResponse::InvalidDnSyntax:
380 
381                     throw new IncorrectParameterException( "The DN was of incorrect syntax: $dn." );
382                     break;
383 
384                 default:
385 
386                     throw new LdapNativeException( $link_id );
387                     break;
388             }
389         }
390 
391         // return the resource result_id
392         return $result_id;
393     }
