 29 /**
 30  * Performs lookup operations on ldap server
 31  *
 32  * You can use Task to perform searches on the ldap server according to options you
 33  * specify.<br>
 34  * This class serves as a wrapper for php functions that retrieve and process
 35  * data from directory servers.
 36  *
 37  * <h2>Features:</h2>
 38  * <h3>Method Chaining</h3>
 39  * Methods may be chained together to simplify configuration. Ensure that the methods you wish to chain
 40  * together return the {@link self} object to avoid issues. See the example code below.<br>
 41  * <br>
 42  * <p class="alert">Make sure you import the class into current namespace before trying to use it:<br>
 43  * <code>use ADX\Core\Task;</code><br>
 44  * To simplify the examples, the above line is not present in the examples but it is assumed you have
 45  * it in your implementation.</p>
 46  *
 47  * <h2>Example:</h2>
 48  * <code>
 49  * use ADX\Enums;                           // Import the Enums namespace
 50  *
 51  * $link = new Link( 'example.com' );               // Connect to server
 52  * $link->bind( 'user@example.com', 'MySecretPwd' );        // Authenticate
 53  *
 54  * $task = new Task( Enums\Operation::OpSearch, $link );        // Create a search operation
 55  * $task    ->attributes( ['cn', 'givenname', 'sn', 'mail'] )   // Load these attributes for the objects
 56  *  ->filter( '(&(objectclass=user)(mail=*))' )     // Get all users that have a mail attribute
 57  *  ->base( 'OU=Users,DC=example,DC=com' );         // Limit the search only to the Users OU in the domain
 58  *
 59  * // Get the results!
 60  * $result = $task->run();
 61  * </code>
 62  *
 63  * @see         {@link Link} - Use the Link object to establish and configure connection to directory servers
 64  * @see         {@link Enums\Operation} - Specify the scope of the lookup operation
 65  * @see         <a href="http://msdn.microsoft.com/en-us/library/windows/desktop/ms678001%28v=vs.85%29.aspx">MSDN - Deciding Where To Search</a>
 66  *
 67  * @todo        Move all lookup error checking from {@link Object} to {@link self}
 68  * @todo        The connection pooling feature might need a rewrite - if I start a lookup operation with an account that has lower access rights than the one I used previously, AD-X might fail to perform the requested operation.
 69  */

122     /**
123      * Pull changes from Active Directory for a given query
124      *
125      * Use this method to set up a search query and later get all changed objects
126      * for that query. This method returns a special {@link Delta} object that preserves
127      * your search configuration and you use this object for consequent
128      * searches. It also contains your search results. When you run the method for the first time,
129      * it behaves as if you performed just an ordinary search, but on consequent searches,
130      * it only returns objects that were modified or deleted since you last ran the search query.
131      *
132      * <br>
133      *
134      * <p class='alert'>Note that due to the way this feature is implemented it is impossible to
135      * search for objects where only specific attributes have been modified - the method returns all objects
136      * that were modified in any way since your last execution and it is your task to determine if
137      * the changes happened on attributes that you are interested in.</p>
138      * <br>
139      *
140      * This method takes two possible forms of arguments:<br>
141      * When you run it for the first time, you pass the search parameters
142      * as for any other ldap search query - the search filter, the list of attributes to be
143      * returned and a fully configured {@link Link} instance.
144      *
145      * <br>
146      *
147      * For all consecutive executions, you only pass the {@link Delta} instance
148      * that you got when you ran the method for the first time and, again, a configured {@link Link}
149      * instance.
150      *
151      * <h2>Example:</h2>
152      * To watch for changes on all users in your domain:
153      * <code>
154      * // Establish connection to your directory server
155      * $link = new Link( 'example.com' );
156      * $link->bind( 'admin@example.com', 'SecretPwd' );
157      *
158      * // When called for the first time for a particular query, you get
159      * // an instance of Delta class - this instance contains the results
160      * // of your query as well as all the information/state necessary
161      * // to perform subsequent queries to get all changes for that resultset
162      * $delta = Task::changes( '(objectcategory=person)', ['mail', 'name', 'memberof'], $link );
163      * $result = $delta->result;    // Here's where the actual data is
164      * // Do something with result...
165      *
166      * // Now it's time to store the $delta instance someplace safe so we can get
167      * // changes for this query at a later time - you should serialise the instance
168      * // and store it somewhere - i.e. in a database or on disk
169      * file_put_contents( 'my_query_state.txt', serialize( $delta ) );  // Always serialise!
170      *
171      * // Many moments pass...
172      * // In another script, far, far away...
173      *
174      * // It's time to see which objects have been modified since our last run
175      * // Connect to ldap again
176      * $link = new Link( 'example.com' );
177      * $link->bind( 'admin@example.com', 'SecretPwd' );
178      *
179      * // Get the previous query state from the file
180      * $delta = unserialize( file_get_contents( 'my_query_state.txt' ) );
181      * // And get all the objects that have been changed since we last
182      * // run the changes method, but this time we only pass in
183      * // the instance of Delta class - it contains all the information
184      * // about our previous query so we don't have to configure it again
185      * $delta = Task::changes( $delta, $link );
186      * $result = $delta->result;    // Changed objects are here again!
187      *
188      * // Notice that now we have a new instance of the Delta class in the
189      * // $class variable - reflecting the fact that we just ran the query again
190      *
191      * // And so the story continues on and on...
192      * </code>
193      *
194      * <h2>Things you should know</h2>
195      * <ul>
196      * <li>You must always connect to the same domain controller for a particular query,
197      * otherwise you will receive the full resultset on each run</li>
198      * <li>To track deleted objects, all returned objects have an <i>isDeleted</i>
199      * {@link Attribute} automatically that you can use to check if that object has been deleted</li>
200      * <li>You cannot use BaseDN overrides with this feature at this time
201      * ( this might be implemented in the future )</li>
202      * </ul>
203      *
204      * @return      Delta       Object containing the state and data of the changes
205      *
206      * @see         <a href="http://msdn.microsoft.com/en-us/library/ms677627.aspx">MSDN - Polling for changes using usnChanged</a>
207      */
208     public static function changes()
209     {
210         $args       = func_get_args();
211         $link       = array_pop( $args );       // Link is always last
212         $rootDSE    = $link->rootDSE;
213         $server     = $rootDSE->dnsHostName(0); // We will only be able to get a diff when talking to the same server next time
214 
215         if ( $args[0] instanceof Delta )        // This is a continuation of previous diff run
216         {
217             $last_delta     = $args[0];
218             $filter         = $last_delta->filter;
219             $attributes     = $last_delta->attributes;
220             $boundaryUSN    = $last_delta->server == $server ? $last_delta->cookie + 1 : 0; // Pull all data if we are dealing with a different server
221 
222             // Include deleted objects in the resultset
223             $link->show_deleted( true );        // Also make this control extension critical
224         }
225         else                                    // This is a new diff run
226         {
227             $filter         = $args[0];
228             $attributes     = $args[1];
229             $boundaryUSN    = 0;
230         }
231 
232         $task = new static( Enums\Operation::OpSearch, $link );
233         $task->filter( q::a( "(usnChanged>=$boundaryUSN)", $filter ) )  // Add the USN magic to the search query
234              ->attributes( array_merge( $attributes, ['isDeleted'] ) ); // Include isDeleted to help identifying deleted objects
235 
236         // Run, Forest, run!
237         $result = $task->run_paged();
238 
239         return new Delta( $result, $filter, $attributes, $server, $rootDSE->highestCommittedUSN(0) );
240     }

270      * Enable the results to be retrieved from directory server in pages
271      *
272      * This allows you to retrieve complete search results for queries that would otherwise
273      * hit the maximum limit for a single search ( Active Directory limits a single non-paged search
274      * to 1000 objects by default ). Enable this functionality by invoking this
275      * method on the Task and perform search queries using the same search criteria

277      * by invoking {@link Task::get_all_pages()} and you will get all the data with a single command.
278      *
279      * @uses        Enums\ServerControl::PagedResults
280      * @param       int     Number of objects in a single page ( default is 1000 )
281      * @param       string      Base64-encoded pagination cookie to be re-used. If not specified, an empty cookie will be used
282      * @return      self
283      *
284      * @see         <a href="http://www.php.net/manual/en/function.ldap-control-paged-result.php">PHP - ldap_control_paged_result()</a>

367     /**
368      * Perform the lookup operation on server and return the resultset
369      *
370      * This method sends the lookup request to the server with the
371      * configuration you provided. If the {@link Link} has pagination enabled,
372      * it will return a single page. To get the subsequent pages, call this method
373      * again on the object, without modifying the lookup criteria.
374      *
375      * @uses        Link::$rootDSE      to get the domain base dn ( from <i>defaultnamingcontext</i> ) if no override has been specified
376      * @return      Result|false        The Result object with returned Objects or FALSE if maximum number of referrals was chased
377      */
378     public function run()
379     {
380         // Prepare the information needed for the operation
381         $link_id    = $this->adxLink->get_link();
382         $baseDN     = ( $this->dn || $this->dn === '' ) ? $this->dn : $this->adxLink->rootDSE->defaultnamingcontext(0);
383         if ( $this->page_size && $this->complete ) $this->cookie = '';  // Reset the cookie if the operation is started again
384         $this->complete = false;                                        // Reset the completion status
385 
386         // Send the pagination control cookie if present
387         // I must check explicitly for NULL bcause '' also evaluates to FALSE
388         if ( $this->cookie !== null ) ldap_control_paged_result( $link_id, $this->page_size, true, $this->cookie );
389 
390         // Perform the lookup operation
391         // All exceptions thrown here will bubble up!
392         $result_id = $this->_directory_lookup( $this->operationType, $link_id, $baseDN, $this->filter, $this->attributes );
393 
394         // Get the new cookie from the response if pagination is enabled
395         // I must check explicitly for NULL bcause '' also evaluates to FALSE
396         if ( $this->cookie !== null ) ldap_control_paged_result_response( $link_id, $result_id, $this->cookie );
397 
398         // Lookup operation successful - continue processing...
399         $referral_link = $this->_parse_referrals( $result_id );
400 
401         // Check for referrals
402         if ( $referral_link )                                       // We have been referred to a new Link
403         {
404             if ( $this->referral_counter <= $this->max_referrals )  // Are we still allowed to chase referrals?
405             {
406                 // Do we have this connection in the pool already?
407                 if ( isset( static::$connection_pool[$referral_link] ) )
408                 {
409                     $link = static::$connection_pool[$referral_link];   // Use the existing link to perform the lookup
410                 }
411                 else
412                 {
413                     $link = $this->adxLink->_redirect( $referral_link );    // Reconnect to the new server
414                     static::$connection_pool[$referral_link] = $link;       // Save the connection in the link pool
415                 }
416 
417                 $this->referral_counter++;
418                 $this->adxLink = $link;                                     // Replace the current link with the new link
419 
420                 return $this->run();    // Only those calls that actually succeed in getting data should continue with code below
421             } else return false;        // Reached maximum number of referrals -> do not continue
422         }
423 
424         // Successfully retrieved data from server
425         $this->referral_counter = 0;
426 
427         $result = ldap_get_entries( $this->adxLink->get_link(), $result_id );
428 
429         unset( $result['count'] );  // Clean up some unneeded mess ( more cleanup happens at object instantiation )
430 
431         $data = array();
432 
433         // Create new objects from result, converting the values to php-compatible data formats on the way
434         foreach ( $result as $objectData ) $data[] = new Object( $this->adxLink, $objectData, true );
435 
436         $this->complete = ! $this->cookie;  // As long as cookie is present, the result cannot be complete
437 
438         return new Result( $data );
439     }

441     /**
442      * Return all pages at once when doing paged search operations
443      *
444      * Use this method to get a complete resultset with all pages at once.
445      *
446      * @param       int         Optional size of objects per single page
447      * @return      Result      A Result object containing the objects on the server
448      *

451     public function run_paged( $page_size = null )
452     {
453         isset( $page_size ) ? $this->use_pages( $page_size ) : $this->use_pages();
454 
455         $resultset = array();
456 
457         do
458         {
459             $data = $this->run();
460 
461             if ( $data instanceof Result )
462             {
463                 $resultset = array_merge( $resultset, $data->to_array() );
464             }
465             else return false;
466         }
467         while ( ! $this->complete );
468 
469         return new Result( $resultset );
470     }

473     protected function _parse_referrals( $result_id )
474     {
475         if ( ! ldap_parse_result( $this->adxLink->get_link(), $result_id, $code, $matchedDN, $errMsg, $referrals ) ) return;
476 
477         if ( count( $referrals ) == 0 ) return;
478 
479         // Referral looks like this:
480         // ldap://example.com/DC=example,DC=com
481         //
482         // Extract the protocol and url from the referral string
483         if ( preg_match( "/^ldaps?:\/\/[a-zA-Z0-9_\-.]*/", $referrals[0], $matches ) !== 1 ) throw new Exception( "Unable to parse referral - {$referrals[0]}" );
484 
485         // Match found? Return it!
486         if ( isset( $matches[0] ) ) return $matches[0];
487     }

490     protected function _directory_lookup( $operation, $link_id, $dn, $filter, $attributes )
491     {
492         // Ensure that attributes required by the library to work correctly are always present in the resultset
493         $mandatory = [
494             'objectguid',
495             'objectclass',
496         ];
497 
498         $attributes = array_merge( $attributes, $mandatory );
499 
500         // php's internal mechanisms should take care of that... Keeping this here only for
501         // a possible future reference.
502         // $attributes = array_map( 'strtolower', $attributes );
503         // $attributes = array_unique( $attributes );
504 
505         // Perform the operation
506         $result_id = $operation( $link_id, $dn, $filter, $attributes );
507 
508         if ( $result_id === false ) // Operation was unsuccessful
509         {
510             // Check for any possible errors and handle them ( probably a TODO )
511             $code = ldap_errno( $link_id );
512 
513             switch( $code )
514             {
515                 case Enums\ServerResponse::InvalidDnSyntax:
516 
517                     throw new IncorrectParameterException( "The DN was of incorrect syntax: $dn." );
518                     break;
519 
520                 default:
521 
522                     throw new LdapNativeException( $link_id );
523                     break;
524             }
525         }
526 
527         // return the resource result_id
528         return $result_id;
529     }
