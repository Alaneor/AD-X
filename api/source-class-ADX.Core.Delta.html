 25 /**
 26  * A helper class returned upon retrieving changes from the directory server
 27  *
 28  * Instances of this class represent a current state of a search query for
 29  * which you would later like to retrieve changes. You should always
 30  * serialize these instances and store them for later use in order for the
 31  * change tracking to work.
 32  *
 33  * @property-read       Result      $result         The data returned by the current search ( not set on unserialised instances )
 34  * @property-read       string      $filter         The filter used for the search
 35  * @property-read       array       $attributes     Attributes that you requested to be loaded from server for the returned objects
 36  * @property-read       string      $server         The server name that this search query was run on
 37  *
 38  * @see     {@link Task::changes()}     Pull changes from Active Directory for a given query
 39  */
 40 class Delta
 41 {
 42     protected $result       = null; // The Result object containing the changed objects in an instance of the Result class
 43     protected $filter       = '';   // The filter used for the diff
 44     protected $attributes   = [];   // The attributes that the objects in the diff should have
 45     protected $server       = '';   // The directory server this delta was pulled from
 46     protected $cookie       = '';   // The highestCommittedUsn value of the server
 47 
 48 
 49     /**
 50      * Create a new instance of the Delta class
 51      *
 52      * @param       Result      The data returned by the current search
 53      * @param       string      The filter used for the search
 54      * @param       array       Attributes that are loaded from the server for the returned objects
 55      * @param       string      The server name that this search query was run on
 56      * @param       mixed       A piece of data that keeps track of the state of changes
 57      *
 58      * @internal
 59      */
 60     public function __construct( Result $result, $filter, $attributes, $server, $cookie )
 61     {
 62         $this->result       = $result;
 63         $this->filter       = $filter;
 64         $this->attributes   = $attributes;
 65         $this->server       = $server;
 66         $this->cookie       = $cookie;
 67     }
 68 
 69     /**
 70      * Mapper to provide read-only access to protected properties
 71      *
 72      * @internal
 73      */
 74     public function __get( $property )
 75     {
 76         if ( ! isset( $this->$property ) || $property == 'cookie' )
 77         {
 78             $trace = debug_backtrace();
 79             trigger_error(
 80                 'Undefined read-only property: ' . $property .
 81                 ' in ' . $trace[0]['file'] .
 82                 ' on line ' . $trace[0]['line'],
 83                 E_USER_NOTICE );
 84 
 85             return null;
 86         }
 87 
 88         return $this->$property;
 89     }
 90 
 91     /**
 92      * Clean up the object a bit before serialisation
 93      *
 94      * Just get rid of any resultsets present on the instance -
 95      * there's no reason to have all that data serialised
 96      *
 97      * @internal
 98      */
 99     public function __sleep()
100     {
101         return [
102             'filter',
103             'attributes',
104             'server',
105             'cookie',
106         ];
107     }
108 }
