 25 /**
 26  * This query builder provides a means of creating ldap search filters in a simple way
 27  *
 28  * Use this class to build your ldap search filter. Current support is implemented
 29  * for the AND( & ), OR( | ) and NOT( ! ) operators.
 30  *
 31  * Usage of this class is completely optional - you can either use it to build
 32  * your filter or supply your own search filter as a regular string.
 33  *
 34  * <h2>Example:</h2>
 35  * <code>
 36  * use ADX\Core\Query;
 37  *
 38  * // The simplest filter - all objects with email address present:
 39  * $filter = Query::a(
 40  *  [ 'mail' => '*' ]
 41  * );
 42  * // (&(mail=*))
 43  *
 44  * // Limit the objects to only user accounts with email address present:
 45  * $filter = Query::a([
 46  *  'objectclass'   => 'user',
 47  *  'mail'      => '*'
 48  * ]);
 49  * // (&(objectclass=user)(mail=*))
 50  *
 51  * // All users that do NOT have the email address set:
 52  * // ( notice that we pass multiple parameters to the
 53  * // Query::a() method - you can pass as many as you like -
 54  * // as long as they are hashes or strings )
 55  * $filter = Query::a(
 56  *  ['objectclass' => 'user'],
 57  *  Query::n([ 'mail' => '*' ]) // We pass multiple parameters here to the Query::a() method
 58  * );
 59  * // (&(objectclass=user)(!(mail=*)))
 60  * </code>
 61  */

 65      * Build a logical AND filter - (&(attribute=value))

 67      * @param   array|string    $data,...   Unlimited number of hash arrays or strings to be used

119     /**
120      * Loop through the arguments and generate a proper ldap filter using the provided operator
121      *
122      * @param   array   Array containing the attribute => value mappings, or a string with already generated ldap filter
123      * @return  string  The generated ldap filter
124      */
125     protected static function _parse( $args, $operator )
126     {
127         $filter = '';
128 
129         foreach ( $args as $argument )
130         {
131             if ( is_array( $argument ) )
132             {
133                 $result = array_map('static::_stringify', array_keys( $argument ), $argument );
134 
135                 $filter .= implode( '', $result );
136             }
137             else $filter .= $argument;
138         }
139 
140         return "($operator$filter)";
141     }

143     /**
144      * Encapsulate the key and value pair in a proper ldap filter string
145      *
146      * @param   string      The ldap property to be filtered for
147      * @param   string      The value to search for
148      * @param   string      Optional logic to be used ( Bitwise or Extended match filters )
149      *
150      * @return  string      The formatted ldap filter
151      */
152     protected static function _stringify( $key, $value, $logic = null )
153     {
154         return "($key" . ( $logic ? ":$logic:" : "" ) . "=$value)";
155     }
