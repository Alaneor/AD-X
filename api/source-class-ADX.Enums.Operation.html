205 // This exception is thrown when the php's ldap library emits
206 // error messages ( where taken care of in the code )
207 class LdapNativeException extends Exception
208 {
209     public function __construct( $link_id )
210     {
211         parent::__construct();
212 
213         ldap_get_option( $link_id, LDAP_OPT_HOST_NAME, $domain );
214 
215         $domain         = explode( ':', $domain )[0]; // Get the server name and discard the port
216         $this->code     = ldap_errno( $link_id );
217         $this->message  = "$domain: ".ldap_error( $link_id );
218     }
219 }
